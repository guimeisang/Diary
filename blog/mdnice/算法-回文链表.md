#### 题目
```js
1 > 2 > 2 > 1
return true

1 > 2 > 2

/**
 * @param {ListNode} head
 * @return {boolean}
 * singly-linked list
 * function ListNode(val, next) {
 *  this.val = (val === undefined ? 0 : val)
 *  this.next = (next === undefined ? null : next)
 * } 
 */
var isPalindrome = function(head) {

}
```

#### 解题
```
var isPalindrome = function(head) {
  let a = '', b = ''
  while(head != null) {
    a = a + head.val
    b = head.val + b
    head = head.next
  }

  return a === b
}
```

1. 通过这种方便比较巧妙的解决，并且复杂度是o(n)
